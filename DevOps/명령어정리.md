```bash
# 파일이나 디렉토리의 권한 변경
chmod [옵션] [권한] [파일 또는 디렉토리]

# 예시: 파일에 소유자에게 읽기, 쓰기, 실행 권한을 부여합니다.
chmod u+rwx [파일]

# 예시: 파일에 그룹에게 읽기 권한을 부여합니다.
chmod g+r [파일]

# 예시: 파일에 다른 사용자에게 읽기, 쓰기, 실행 권한을 제거합니다.
chmod o-rwx [파일]

# 예시: 파일에 모든 사용자에게 실행 권한을 부여합니다.
chmod a+x [파일]

# 파일이나 디렉토리의 소유자 변경
chown [옵션] [소유자:그룹] [파일 또는 디렉토리]

# 예시: 파일의 소유자를 user로, 그룹을 group으로 변경합니다.
chown user:group [파일]

# 예시: 현재 사용자가 소유한 파일에 대한 권한을 변경합니다.
sudo chmod u+rwx [파일]
```

```bash
# 현재 디렉토리의 파일 목록 출력
ls

# 디렉토리 변경
cd [디렉토리 경로]

# 새로운 디렉토리 생성
mkdir [디렉토리 이름]

# 파일 생성
touch [파일 이름]

# 파일 삭제
rm [파일 이름]

# 디렉토리 삭제
rm -r [디렉토리 이름]

# 파일 복사
cp [소스 경로] [대상 경로]

# 디렉토리 복사
cp -r [소스 경로] [대상 경로]

# 파일 이동
mv [소스 경로] [대상 경로]

# 디렉토리 이동
mv [소스 경로] [대상 경로]

# 파일 내용 출력
cat [파일 이름]

# 파일 내용 검색
grep [검색어] [파일 이름]

# 파일 편집
nano [파일 이름]

# 사용자 추가
adduser [사용자 이름]

# 사용자 삭제
deluser [사용자 이름]

# 패키지 목록 업데이트
apt-get update

# 패키지 설치
apt-get install [패키지 이름]

# 패키지 삭제
apt-get remove [패키지 이름]

# 패키지 업그레이드
apt-get upgrade

# 패키지 검색
apt-cache search [패키지 이름]
```

```docker
# 도커 이미지 다운로드
docker pull 이미지_이름

# 도커 이미지 실행
docker run 이미지_이름

# 실행 중인 도커 컨테이너 목록 확인
docker ps

# 모든 도커 컨테이너 목록 확인
docker ps -a

# 도커 컨테이너 중지
docker stop 컨테이너_ID

# 도커 컨테이너 삭제
docker rm 컨테이너_ID

# 도커 이미지 목록 확인
docker images

# 도커 이미지 삭제
docker rmi 이미지_ID

# 도커 컨테이너 내부 쉘 접속
docker exec -it 컨테이너_ID /bin/bash

# 도커 컨테이너 로그 확인
docker logs 컨테이너_ID

# 도커 이미지 빌드
docker build [OPTIONS] PATH

# 도커 이미지 생성
docker commit [OPTIONS] CONTAINER [REPOSITORY[:TAG]]

# 도커 이미지에 태그 추가
docker tag SOURCE_IMAGE[:TAG] TARGET_IMAGE[:TAG]

# 도커 이미지 tar 아카이브로 저장
docker save [OPTIONS] IMAGE [IMAGE...]

# 도커 이미지 tar 아카이브에서 불러오기
docker load [OPTIONS]

# 도커 컨테이너 실행
docker run [OPTIONS] IMAGE [COMMAND] [ARG...]

# 도커 컨테이너 시작
docker start [OPTIONS] CONTAINER [CONTAINER...]

# 도커 컨테이너 중지
docker stop [OPTIONS] CONTAINER [CONTAINER...]

# 도커 컨테이너 삭제
docker rm [OPTIONS] CONTAINER [CONTAINER...]

# 도커 컨테이너 일시 정지
docker pause CONTAINER [CONTAINER...]

# 도커 컨테이너 일시 정지 해제
docker unpause CONTAINER [CONTAINER...]

# 도커 컨테이너 내부에서 명령어 실행
docker exec [OPTIONS] CONTAINER COMMAND [ARG...]

# 도커 네트워크 목록 확인
docker network ls

# 도커 네트워크 생성
docker network create [OPTIONS] NETWORK

# 도커 네트워크에 컨테이너 연결
docker network connect [OPTIONS] NETWORK CONTAINER

# 도커 네트워크에서 컨테이너 분리
docker network disconnect [OPTIONS] NETWORK CONTAINER

# 도커 볼륨 목록 확인
docker volume ls

# 도커 볼륨 생성
docker volume create [OPTIONS] VOLUME

# 도커 볼륨 상세 정보 확인
docker volume inspect [OPTIONS] VOLUME

# 도커 볼륨 삭제
docker volume rm [OPTIONS] VOLUME [VOLUME...]

# 도커 컨테이너 백그라운드 실행
docker run -d [OPTIONS] IMAGE [COMMAND] [ARG...]

# 도커 컨테이너 호스트와 컨테이너의 포트 연결
docker run -p HOST_PORT:CONTAINER_PORT [OPTIONS] IMAGE [COMMAND] [ARG...]

# 도커 컨테이너에서 환경변수 설정
docker run -e KEY=VALUE [OPTIONS] IMAGE [COMMAND] [ARG...]

# 도커 컨테이너에서 로컬 디렉토리와 볼륨 연결
docker run -v LOCAL_DIR:CONTAINER_DIR [OPTIONS] IMAGE [COMMAND] [ARG...]

# 도커 컨테이너에서 호스트와 볼륨 연결
docker run -v HOST_DIR:CONTAINER_DIR [OPTIONS] IMAGE [COMMAND] [ARG...]

# 도커 컨테이너에서 컨테이너 간 통신
docker run --link NAME_OR_ID [OPTIONS] IMAGE [COMMAND] [ARG...]

# 도커 컨테이너에서 호스트와 시간 동기화
docker run --privileged [OPTIONS] IMAGE [COMMAND] [ARG...]

# 도커 컨테이너에서 사용할 CPU 개수 설정
docker run --cpus=1.5 [OPTIONS] IMAGE [COMMAND] [ARG...]

# 도커 컨테이너에서 사용할 메모리 양 설정
docker run -m 512m [OPTIONS] IMAGE [COMMAND] [ARG...]

# 도커 이미지 레지스트리에 업로드
docker push [OPTIONS] NAME[:TAG]

# 도커 이미지 레지스트리에서 다운로드
docker pull [OPTIONS] NAME[:TAG]

# 도커 컨테이너와 호스트 간 파일 복사
docker cp [OPTIONS] SRC_PATH CONTAINER:DEST_PATH
docker cp [OPTIONS] CONTAINER:SRC_PATH DEST_PATH

# 도커 네트워크 드라이버 설정
docker network create -d bridge NETWORK
docker network create -d host NETWORK
docker network create -d overlay NETWORK
docker network create -d macvlan NETWORK

# 도커 컨테이너 로그 출력 옵션
docker logs [OPTIONS] CONTAINER

# 도커 컨테이너 환경변수 출력
docker inspect --format='{{range $e := .Config.Env}}{{$e}} {{end}}' CONTAINER

# 도커 컨테이너 정보 출력
docker inspect [OPTIONS] CONTAINER

# 도커 컨테이너 로그 파일 위치
```
